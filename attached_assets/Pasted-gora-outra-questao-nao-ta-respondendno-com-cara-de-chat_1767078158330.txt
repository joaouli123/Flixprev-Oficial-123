// ============================================
// 1️⃣ RESPONSE ORCHESTRATOR (VERSÃO HUMANIZADA)
// ============================================

function orchestrateResponse(rawResponse, questionType, hasContext = true) {
  // Se não tem contexto, mantém a resposta negativa padrão (que já tem sugestão)
  if (!hasContext || !rawResponse || rawResponse.trim().length === 0) {
    return getRandomNegativeResponse();
  }

  let formattedResponse = rawResponse;

  // Removemos a obrigatoriedade de "anchorPhrases" (Frases de Ancoragem)
  // Deixamos o LLM (GPT) decidir como começar a frase.
  
  // Apenas garantimos formatação visual se for lista
  if (questionType === 'factual' && (rawResponse.includes('•') || rawResponse.includes('- '))) {
     formattedResponse = ensureProperListFormat(rawResponse);
  }

  // Se quiser manter a citação [Trecho ID], pode deixar, pois dá credibilidade.
  // Mas removemos o texto introdutório forçado.

  return formattedResponse;
}
