CREATE TABLE document_chunks (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  document_id UUID REFERENCES documents(id) ON DELETE CASCADE,
  content TEXT NOT NULL,
  embedding VECTOR(3072)
);

Extrair texto do PDF:
import fs from "fs";
import pdf from "pdf-parse";

export async function extractPdfText(path) {
  const buffer = fs.readFileSync(path);
  const data = await pdf(buffer);
  return data.text;
}

Chunking correto:
export function chunkText(text, size = 800, overlap = 150) {
  const chunks = [];
  let start = 0;

  while (start < text.length) {
    chunks.push(text.slice(start, start + size));
    start += size - overlap;
  }

  return chunks;
}

🧠 4️⃣ Gerar embeddings (OpenAI)
import OpenAI from "openai";
const openai = new OpenAI();

export async function embedChunks(chunks) {
  const embeddings = [];

  for (const chunk of chunks) {
    const res = await openai.embeddings.create({
      model: "text-embedding-3-large",
      input: chunk
    });
    embeddings.push(res.data[0].embedding);
  }

  return embeddings;
}

💾 5️⃣ Salvar no Postgres
import { pool } from "./db.js";

export async function saveChunks(documentId, chunks, embeddings) {
  for (let i = 0; i < chunks.length; i++) {
    await pool.query(
      `
      INSERT INTO document_chunks (document_id, content, embedding)
      VALUES ($1, $2, $3)
      `,
      [documentId, chunks[i], embeddings[i]]
    );
  }
}

Node:
export async function searchChunks(queryEmbedding) {
  const res = await pool.query(
    `
    SELECT content
    FROM document_chunks
    ORDER BY embedding <-> $1
    LIMIT 5
    `,
    [queryEmbedding]
  );

  return res.rows.map(r => r.content);
}

🤖 7️⃣ Prompt HARD-RESTRICTED (CRÍTICO)
export function buildPrompt(context, question) {
  return `
Você é um assistente que responde EXCLUSIVAMENTE com base no texto abaixo.

REGRAS OBRIGATÓRIAS:
- Use apenas o conteúdo fornecido
- NÃO use conhecimento externo
- NÃO invente autores, conceitos ou práticas
- Se a resposta não estiver no texto, responda exatamente:
  "O documento não aborda esse ponto."

DOCUMENTO:
${context.join("\n\n---\n\n")}

PERGUNTA:
${question}
`;
}

🧪 8️⃣ Endpoint de pergunta (Node)
app.post("/ask", async (req, res) => {
  const { question } = req.body;

  const embeddingRes = await openai.embeddings.create({
    model: "text-embedding-3-large",
    input: question
  });

  const queryEmbedding = embeddingRes.data[0].embedding;

  const chunks = await searchChunks(queryEmbedding);

  const prompt = buildPrompt(chunks, question);

  const completion = await openai.chat.completions.create({
    model: "gpt-4.1-mini",
    messages: [{ role: "user", content: prompt }],
    temperature: 0
  });

  res.json({ answer: completion.choices[0].message.content });
});
